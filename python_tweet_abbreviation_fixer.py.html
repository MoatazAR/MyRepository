my_dict = {'u':'you','idk':'I do not know','omw':'on my way','r':'are','tbh':'to be honest','abt':'about','b4':'before','btw':'by the way','lmk':'let me know', 'c':'see','ur':'your','u\'r':'you are','asap':'as soon as possible','isn\'t':'is not','aren\'t':'are not','wasn\'t':'was not','can\'t':'can not','don\'t':'do not', '2':'to', 'we\'ll':'we will','who\'s':'who is','brb':'be right back','jk':'just kidding','thx':'thanks','i\'m':'I am','we\'re':'we are','i\'ve':'I have','ppl':'people','you\'re':'you are','gonna':'going to','plz':'please'}
#program responsible for taking a file of tweets and creatign a new file with the corrected version of the tweets (changes abbreviations to full form)
corrected_tweets = []
tweet_file = open("tweets.txt","r")
file_lines = tweet_file.readlines()
for line in file_lines:
    my_tweet = line
    my_tweet = my_tweet.replace(",", " ,")
    my_tweet = my_tweet.replace(".", " .")
    my_tweet = my_tweet.replace("?", " ?")
    words = my_tweet.split()
    i = 0
    j = len(words)
    for i in range(j):
        if words[i].lower() in my_dict:
            words[i]=my_dict[words[i].lower()]
    words[0]=words[0].capitalize()
    final_tweet = ' '.join(words)
    final_tweet = final_tweet.replace(" ,", ",")
    final_tweet = final_tweet.replace(" .", ".")
    final_tweet = final_tweet.replace(" ?", "?")
    corrected_tweets.append(final_tweet)
    #print(final_tweet)
tweet_file.close()
new_tweets = open("new_tweets.txt","w")
for tweet in corrected_tweets:
    new_tweets.write("\n"+tweet)
